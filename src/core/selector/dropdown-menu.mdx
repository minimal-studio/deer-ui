# DropdownMenu

> 基于 selector 的下拉选择器组件，支持多选

<PropsTable of={DropdownMenu} />

## 实例

<Playground>
  {
    () => {
      const values = {
        value1: '显示的值1',
        value2: '显示的值2',
        value3: '显示的值3',
      }
      const values2 = {
        0: '数字0',
        1: '数字1',
        2: '数字2',
        3: '数字3',
      }
      let dropdown, dropdownMul;
      return (
        <State initial={{
          value: 'value1',
          valueMul: [],
          valueMul2: [],
        }}>
          {
            ({ state, setState }) => {
              const { value, valueMul, valueMul2 } = state;
              return (
                <div>
                  <div className="mu10">
                    <DropdownMenu
                      ref={e => dropdown = e}
                      onChange={val => {
                        setState({
                          value: val
                        })
                      }}
                      value={value}
                      values={values}
                    />
                    <span className="ml10">单选实际的值 {value}</span>
                  </div>
                  <div className="mu10">
                    <DropdownMenu
                      ref={e => dropdownMul = e}
                      onChange={val => {
                        setState({
                          valueMul: val
                        })
                      }}
                      isMultiple={true}
                      values={values}
                    />
                    <span className="ml10">多选的实际的值 {valueMul.join(',')}</span>
                  </div>
                  <div className="mu10">
                    <DropdownMenu
                      ref={e => dropdownMul = e}
                      onChange={val => {
                        setState({
                          valueMul2: val
                        })
                      }}
                      isMultiple={true}
                      isNum={true}
                      defaultValue={[1, 0]}
                      values={values2}
                    />
                    <span className="ml10">多选的实际的值 {valueMul2.join(',')}</span>
                  </div>
                  <div>
                    <div className="mu10">可以通过 ref 来改变 DropdownMenu 的值</div>
                    <div className="btn theme" onClick={e => dropdown.changeValue('value3')}>点击改变为 value3</div>
                  </div>
                  <hr/>
                  <div>
                    <span className="ms10">向上左弹出</span>
                    <DropdownMenu
                      position="top,left"
                      onChange={val => {
                        setState({
                          value: val
                        })
                      }}
                      value={value}
                      values={values}
                    />
                    <hr/>
                    <span className="ms10">向上右弹出</span>
                    <DropdownMenu
                      position="top,right"
                      onChange={val => {
                        setState({
                          value: val
                        })
                      }}
                      value={value}
                      values={values}
                    />
                    <hr/>
                    <span className="ms10">无效的值示例，这个时候需要检查参数</span>
                    <DropdownMenu
                      position="top,right"
                      onChange={val => {
                        setState({
                          value: val
                        })
                      }}
                      value={'123'}
                      values={values}
                    />
                  </div>
                </div>
              )
            }
          }
        </State>
      )
    }
  }
</Playground>
