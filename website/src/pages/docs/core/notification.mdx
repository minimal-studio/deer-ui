# Notify

> 用于全局重要通知

## Basic Usage

```jsx live noInline
import { Notify, CancelNotify, NotifyEvent } from '@deer-ui/core';

function Example() {
  return (
    <>
      <span className="btn theme mr10"
        onClick={e => {
          Notify({
            position: 'top,right',
            text: '通知内容',
            // id: 'ID',
            title: '通知标题',
            type: 'white',
            timer: -1,
            onClickTip: (e) => {console.log(e)},
            actionText: '有 action 时显示的文字',
          })
        }}>
        广播通知
      </span>
      <span className="btn red mr10"
        onClick={e => {
          Notify({
            position: 'bottom,right',
            text: '通知内容',
            // id: 'ID',
            title: '通知标题',
            type: 'warn',
            timer: 7,
            onClickTip: (e) => {console.log(e)},
            actionText: '有 action 时显示的文字',
          })
        }}>
        右下角广播通知
      </span>
      <span className="btn green mr10"
        onClick={e => {
          Notify({
            position: 'bottom,right',
            text: '通知内容',
            // id: 'ID',
            title: '通知标题',
            type: 'success',
            timer: 7,
            onClickTip: (e) => {console.log(e)},
            actionText: '有 action 时显示的文字',
          })
        }}>
        成功
      </span>
      <span className="btn yellow mr10"
        onClick={e => {
          Notify({
            position: 'top,left',
            text: '通知内容',
            // id: 'ID',
            title: '通知标题',
            type: 'success',
            timer: 0,
            onClickTip: (e) => {console.log(e)},
            actionText: '有 action 时显示的文字',
          })
        }}>
        左上角
      </span>
    </>
  )
}
render(<Example />);
```

## 通过 EventEmitter 广播

```jsx live noInline
import { EventEmitter } from '@mini-code/base-func';
import { NotifyEvent } from '@deer-ui/core';

function Example() {
  return (
    <span className="btn theme mr10"
      onClick={e => {
        EventEmitter.emit(NotifyEvent, {
          text: '通过 EventEmitter 广播通知',
          title: '通知标题',
          type: 'success',
          timer: 7,
          onClickTip: (e) => {console.log(e)},
          actionText: '有 action 时显示的文字',
        })
      }}>
      通过 EventEmitter 广播通知
    </span>
  )
}
render(<Example />);
```

## 取消该广播

```jsx live noInline
import { Notify, CancelNotify, NotifyEvent } from '@deer-ui/core';

function Example() {
  let NotifyID
  return (
    <div>
      <span className="btn theme mr10"
        onClick={e => {
          NotifyID = Notify({
            position: 'top,right',
            text: '通知内容',
            // id: 'ID',
            title: '通知标题',
            type: 'white',
            timer: -1,
            onClickTip: (e) => {console.log(e)},
            actionText: '有 action 时显示的文字',
          })
        }}>
        广播通知
      </span>
      <span className="btn red mr10" onClick={e => {
        CancelNotify(NotifyID)
      }}>取消广播</span>
    </div>
  )
}
render(<Example />);
```
